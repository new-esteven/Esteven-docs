func load_data(path):
	var file = FileAccess.open(path, FileAccess.READ)
	var text = file.get_as_text()

	var json = JSON.new()
	json.parse(text)

	var data = json.data
	
	return data

func load_character(number):
	
	characters[number].show()
	get_node("ColorRect/Label").text = str(skills[number][0])
	get_node("ColorRect/Label2").text = str(skills[number][1])
	get_node("ColorRect/Label3").text = str(skills[number][2])
	$ChacLife.text = "HP: " + str(characters[number].hp)
	if number == 0 and character_count != 1:
		characters[character_count-1].hide()
	if number > 0:
		characters[number-1].hide()
	
func shadow_turn():
	if int(get_node("lifeShadow").text) <= 0:
		get_node("bad guys/shadow").hide()
		await end_battle(true, characters[-1].name, true)
		return
	hide_menu()
	await get_tree().create_timer(2.0).timeout
	for shadow in shadows:
		var affected = randi() % character_count
		load_character(affected)
		characters[affected].hp -= attack(("partyStats/"+str(characters[affected].name+"/hp")), 1, shadow.stats["attack"], shadow.stats["endurance"],
			20, true, shadow.stats["luck"], randf())
		$info.text = str(shadow.name)+" just attacked!"
		if get_node("/root/combat/"+str(characters[affected].name)+"_hit"):
				get_node("/root/combat/"+str(characters[affected].name)+"_hit").play()
				get_node("/root/combat/AudioStreamPlayer2D").volume_db -= 10
				await get_node("/root/combat/"+str(characters[affected].name)+"_hit").finished
				get_node("/root/combat/AudioStreamPlayer2D").volume_db += 10
		await get_tree().create_timer(1.0).timeout
		open_general_menu()

func change_turn():
	turn += 1
	if turn > character_count-1:
		battle_busy = true
		$turn.show()
		await shadow_turn()
		turn = 0
		battle_busy = false
		$turn.hide()
	load_character(turn)
	
func attack(shadow, multiplier, base_attack,
 base_defense, skill_power, is_physical_attack, luck, skill):
	# shadow is the node that stores the shadow's life, or label with chac hp
	# DAMAGE FORMULA BULLSHIT
	
	# The core math formula
	var ratio: float = float(base_attack) / float(max(1, base_defense))
	if skill:
		ratio = float(base_attack * (base_attack * skill)) / float(max(1, base_defense))
	var base_damage: float = sqrt(skill_power * ratio) * 5.0

	# Crit roll based entirely on Luck
	var is_critical: bool = false
	if is_physical_attack:
		var crit_chance: float = luck * 0.005
		if randf() <= crit_chance:
			is_critical = true

	# Apply modifiers
	var modifier: float = 1.0
	if is_critical:
		modifier *= 1.5

	# Random variance (+/- 5%)
	var variance: float = randf_range(0.95, 1.05)

	# Final calculation
	var final_damage: int = floor(base_damage * modifier * variance)
	# continue
	get_node(shadow).text = "HP: "+str(int(get_node(shadow).text) - 
			int(final_damage))
	
	$damage.show()
	$damage.text = str(final_damage)
	return final_damage
	
func play_sound(character):
	if get_node("/root/combat/"+str(character)+"_attack"):
		get_node("/root/combat/"+str(character)+"_attack").play()
		get_node("/root/combat/AudioStreamPlayer2D").volume_db -= 10
		await get_node("/root/combat/"+str(character)+"_attack").finished
		get_node("/root/combat/AudioStreamPlayer2D").volume_db += 10
